<// O Hamming Weight de um número inteiro é dado pela quantidade de bits com valor 1, comment>
<// encontrados na representação binaria deste mesmo inteiro., comment>
<// O algoritmo de Kerningham é uma maneira eficiente de calcular o Hamming Weight,, comment>
<// pois itera apenas k vezes, onde k é o Hamming Weight da entrada., comment>
<// A propriedade-chave a ser compreendida se encontra abaixo, comment>
<// N && (N-1) = M, onde M é a representação binaria de N, com o bit de valor 1 mais significativo, comment>
<// sendo alterado para 0., comment>
<MODULE, main>
<HAMMING_WEIGHT_KERNIGHAN, id>
<(, lparen>
<NAT, type_nat>
<n, id>
<), rparen>
<{, lbrace>
<// count: acumulador que armazena a quantidade de bits ativos (1) encontrados., comment>
<NAT, type_nat>
<count, id>
<←, assign>
<0, nat>
<;, semi>
<// Laço executa enquanto houver pelo menos um bit ativo em n., comment>
<WHILE, while>
<(, lparen>
<n, id>
<>, gt>
<0, nat>
<), rparen>
<DO, do>
<{, lbrace>
<count, id>
<←, assign>
<count, id>
<+, plus>
<1, nat>
<//A linha seguinte reseta um LSB '1' para '0'., comment>
<n, id>
<←, assign>
<n, id>
<&&, bitwise_and>
<(, lparen>
<n, id>
<-, minus>
<1, nat>
<), rparen>
<;, semi>
<}, rbrace>
<RETURN, return>
<count, id>
<;, semi>
<}, rbrace>
