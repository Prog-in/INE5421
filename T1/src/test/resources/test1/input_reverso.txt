COMPRIMENTO(str S)
{
    CHR tmp;
    NAT i <- 0;
    WHILE TRUE
    {
	tmp <- S[i];
	IF (tmp = '\0') THEN
	{
	    BREAK;
	}
	i++;
    }
    RETURN i;
}


TOP_LEVEL INVERTER_CADEIA(str A)
{

    # Quando idx_esquerda >= idx_direita, todos os pares foram trocados.
