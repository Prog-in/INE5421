PROGRAM EUCLID_GCD(NAT a, NAT b)
{
    // Tratando a=0 e b=0
    IF (a = 0) {
	RETURN b;
	}
    IF (b = 0) {
        RETURN a;
	}
    // Caso geral
    WHILE (b != 0) {
        NAT resto <- a % b;
        a <- b;
        b <- resto;
    	}
    // Ao fim do laco a variavel a armazena o GCD
    RETURN a;

}
